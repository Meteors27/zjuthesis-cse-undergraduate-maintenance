% 用于生成翻译部分的开头内容
\newcommand{\translationheader}[4]
{
    {
        {
            % 设置粗体系列
            \bfseries
            % 设置段落缩进为0
            \setlength{\parindent}{0pt}
            % 输出译文题目
            \par \zihao{-3}译文题目：  {#1}
            % 输出原文题目，引用文献中的标题
            \par \zihao{-3}原文题目： \citefield{#2}{title}
            % 输出原文来源，完整引用文献
            \par \zihao{-3}原文来源： \fullcite{#2}
            % 引用文献
            \nocite{#2}
        }
        {
            % 换行
            \newline
            % 输出摘要
            \par \textbf{摘要：}  {#3}
            % 输出关键字
            \par \textbf{关键字：} {#4}
        }
    }
}
